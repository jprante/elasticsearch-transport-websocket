020package org.elasticsearch.websocket.action;
021
022import java.io.IOException;
023import java.util.Map;
024import org.elasticsearch.client.Client;
025import org.elasticsearch.common.inject.Inject;
026import org.elasticsearch.common.settings.Settings;
027import org.elasticsearch.websocket.BaseInteractiveHandler;
028import org.elasticsearch.websocket.InteractiveChannel;
029import org.elasticsearch.websocket.InteractiveController;
030import org.elasticsearch.websocket.InteractiveRequest;
031import org.elasticsearch.websocket.http.HttpServerTransport;
032import org.elasticsearch.websocket.http.netty.NettyInteractiveResponse;
033import org.jboss.netty.channel.Channel;
034
035/**
036 * Forwarding a message to a destination.
037 * 
038 * @author JÃ¶rg Prante <joergprante@gmail.com>
039 */
040public class ForwardAction extends BaseInteractiveHandler {
041
042    private final String TYPE = "forward";
043    private final HttpServerTransport transport;
044
045    @Inject
046    public ForwardAction(Settings settings,
047            Client client,
048            HttpServerTransport transport,
049            InteractiveController controller) {
050        super(settings, client);
051        this.transport = transport;
052        controller.registerHandler(TYPE, this);
053    }
054
055    @Override
056    public void handleRequest(final InteractiveRequest request, final InteractiveChannel channel) {
057        Map<String, Object> m = request.asMap();
058        Integer id = (Integer) m.get("channel");
059        Channel ch = transport.channel(id);
060        try {
061            if (ch != null) {
062                ch.write(new NettyInteractiveResponse("message", (Map<String, Object>) m.get("message")).response());
063                // don't send a success message back to the channel
064            } else {
065                // delivery failed, channel not present
066                channel.sendResponse(TYPE, new IOException("channel " + id + " gone"));
067            }
068        } catch (IOException ex) {
069            logger.error("error while delivering forward message {}: {}", m, ex);
070        }
071    }
072}
