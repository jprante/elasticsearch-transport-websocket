031import org.elasticsearch.common.xcontent.XContentBuilder;
032import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
033import static org.elasticsearch.index.query.FilterBuilders.numericRangeFilter;
034import org.elasticsearch.index.query.NumericRangeFilterBuilder;
035import org.elasticsearch.index.query.QueryBuilder;
036import static org.elasticsearch.index.query.QueryBuilders.termQuery;
037import org.elasticsearch.search.SearchHit;

045/**
046 * Subscribe action. It performs the subscription of a client to
047 * the pubsub index under a given topic.
048 * 
049 * @author JÃ¶rg Prante <joergprante@gmail.com>
050 */
051public class SubscribeAction extends PublishSubscribe {
052
053    protected final static String TYPE = "subscribe";
054
055    @Inject
056    public SubscribeAction(Settings settings,
057            Client client,
058            HttpServerTransport transport,
059            InteractiveController controller,
060            Checkpointer service) {
061        super(settings, client, transport, controller, service);
062        controller.registerHandler(TYPE, this);
063    }
064
065    @Override
066    public void handleRequest(final InteractiveRequest request, final InteractiveChannel channel) {
067        final String topic = request.hasParam("topic") ? request.paramAsString("topic") : "*";
068        final String subscriberId = request.hasParam("subscriber") ? request.paramAsString("subscriber") : null;
069        if (subscriberId == null) {
070            try {
071                channel.sendResponse(TYPE, new IllegalArgumentException("no subscriber"));
072            } catch (IOException e) {
073                logger.error("error while sending failure response", e);
074            }
075        }
076        try {
077            client.prepareIndex()
078                    .setIndex(pubSubIndexName)
079                    .setType(TYPE)
080                    .setId(subscriberId)
081                    .setSource(createSubscriberMessage(topic, channel))
082                    .setRefresh(request.paramAsBoolean("refresh", true))
083                    .execute(new ActionListener<IndexResponse>() {
084                @Override
085                public void onResponse(IndexResponse response) {
086                    try {
087                        XContentBuilder builder = jsonBuilder();
088                        builder.startObject().field("ok", true).field("id", response.id()).endObject();
089                        channel.sendResponse(TYPE, builder);
090                        // receive outstanding messages
091                        sync(subscriberId, topic, channel.getChannel());
092                    } catch (Exception e) {
093                        onFailure(e);
094                    }
095                }
096
097                @Override
098                public void onFailure(Throwable e) {
099                    logger.error("error while processing subscribe request", e);
100                    try {
101                        channel.sendResponse(TYPE, e);
102                    } catch (IOException ex) {
103                        logger.error("error while sending error response", ex);
104                    }
105                }
106            });
107        } catch (Exception e) {
108            logger.error("exception while processing subscribe request", e);
109            try {
110                channel.sendResponse(TYPE, e);
111            } catch (IOException e1) {
112                logger.error("exception while sending exception response", e1);
113            }
114        }
115    }
116
117    private XContentBuilder createSubscriberMessage(String topic, InteractiveChannel channel) {
118        Integer channelId = channel.getChannel().getId();
119        String localAddress = channel.getChannel().getLocalAddress().toString();
120        String remoteAddress = channel.getChannel().getRemoteAddress().toString();
121        try {
122            return jsonBuilder()
123                    .startObject()
124                    .field("topic", topic)
125                    .startObject("subscriber")
126                    .startObject("channel")
127                    .field("id", channelId)
128                    .field("localAddress", localAddress)
129                    .field("remoteAddress", remoteAddress)
130                    .endObject()
131                    .endObject()
132                    .endObject();
133        } catch (IOException e) {
134            return null;
135        }
136    }
137
138    /**
139     * Synchronize the subscriber with the current messages.
140     * 
141     * @param subscriberId
142     * @param topic
143     * @param channel
144     * @throws IOException 
145     */
146    private void sync(final String subscriberId, final String topic, final Channel channel) throws IOException {
147        Long lastSeen = service.checkpointedAt(subscriberId);
148        Long topicSeen = service.checkpointedAt(topic);
149        // if client appearance is later than topic, do not search for any messages
150        if (lastSeen == null || topicSeen == null || lastSeen >= topicSeen) {
151            return;
152        }
153        // message sync - update subscriber with the current timestamp
154        service.checkpoint(subscriberId);
155        service.flushCheckpoint();        
156        // there are unreceived messages, get all outstanding messages since last seen
157        QueryBuilder queryBuilder = termQuery("topic", topic);
158        NumericRangeFilterBuilder filterBuilder = numericRangeFilter("timestamp").gte(lastSeen);
159        SearchResponse searchResponse = client.prepareSearch()
160                .setIndices(pubSubIndexName)
161                .setTypes("publish")
162                .setSearchType(SearchType.SCAN)
163                .setScroll(scrollTimeout)
164                .setQuery(queryBuilder)
165                .setFilter(filterBuilder)
166                .addField("data")
167                .addField("timestamp")
168                .setSize(scrollSize)
169                .execute().actionGet();
170        boolean failed = searchResponse.failedShards() > 0 || searchResponse.timedOut();
171        if (failed) {
172            logger.error("searching for messages for topic {} failed: failed shards={} timeout={}",
173                    topic, searchResponse.failedShards(), searchResponse.timedOut());
174            return;
175        }
176        // look for messages
177        long totalHits = searchResponse.getHits().getTotalHits();
178        boolean zero = totalHits == 0L;
179        if (zero) {
180            return;
181        }
182        // slurp in all outstanding messages
183        while (true) {
184            searchResponse = client.prepareSearchScroll(searchResponse.getScrollId())
185                    .setScroll(scrollTimeout)
186                    .execute().actionGet();
187            for (SearchHit hit : searchResponse.hits()) {
188                long timestamp = hit.field("timestamp").getValue();
189                Map<String, Object> data =  hit.field("data").getValue();
190                channel.write(new NettyInteractiveResponse("message", createPublishMessage(timestamp, data)).response());
191            }
192            if (searchResponse.hits().hits().length == 0) {
193                break;
194            }
195        }
196    }
197    
198    private XContentBuilder createPublishMessage(long timestamp, Map<String,Object> data) {
199        try {
200            return jsonBuilder().startObject()
201                    .field("timestamp", timestamp)
202                    .field("data", data)
203                    .endObject();
204        } catch (IOException e) {
205            return null;
206        }
207    }
