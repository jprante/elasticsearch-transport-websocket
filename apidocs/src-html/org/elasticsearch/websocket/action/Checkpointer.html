019package org.elasticsearch.websocket.action;
020
021import java.io.IOException;
022import org.elasticsearch.ElasticSearchException;
023import org.elasticsearch.action.delete.DeleteRequest;
024import org.elasticsearch.action.get.GetResponse;
025import org.elasticsearch.action.index.IndexRequest;
026import org.elasticsearch.client.Client;
027import org.elasticsearch.client.Requests;
028import org.elasticsearch.common.component.AbstractLifecycleComponent;
029import org.elasticsearch.common.inject.Inject;
030import org.elasticsearch.common.settings.Settings;
031import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
032import org.elasticsearch.websocket.InteractiveChannel;
033
034/*
035 * Checkpoint management for topics and subscribers.
036 * 
037 * @author JÃ¶rg Prante <joergprante@gmail.com>
038 */
039public class Checkpointer extends AbstractLifecycleComponent<Checkpointer> {
040
041    private final String pubSubIndexName;
042    private final Client client;
043    private final static String TYPE = "checkpoint";
044    private final BulkHandler bulkHandler;
045
046    @Inject
047    public Checkpointer(Settings settings, Client client) {
048        super(settings);
049        this.client = client;
050        this.bulkHandler = new BulkHandler(settings, client);
051        this.pubSubIndexName = PubSubIndexName.Conf.indexName(settings);
052    }
053
054    @Override
055    protected void doStart() throws ElasticSearchException {
056    }
057
058    @Override
059    protected void doStop() throws ElasticSearchException {
060    }
061
062    @Override
063    protected void doClose() throws ElasticSearchException {
064    }
065
066    /**
067     * Checkpointing a topic or a subscriber. The current timestamp is written
068     * to the checkpoint index type. Note that bulk index requests are used by
069     * checkpointing and flushCheckpoint() needs to be called after all is done.
070     *
071     * @param id topic or subscriber
072     * @throws IOException
073     */
074    public void checkpoint(String id) throws IOException {
075        indexBulk(Requests.indexRequest(pubSubIndexName).type(TYPE).id(id)
076                .source(jsonBuilder().startObject().field("timestamp", System.currentTimeMillis()).endObject()), null);
077    }
078
079    public void flushCheckpoint() throws IOException {
080        flushBulk(null);
081    }
082
083    public Long checkpointedAt(String id) throws IOException {
084        GetResponse response = client.prepareGet(pubSubIndexName, TYPE, id)
085                .setFields("timestamp")
086                .execute().actionGet();
087        boolean failed = !response.exists();
088        if (failed) {
089            logger.warn("can't get checkpoint for {}", id);
090            return null;
091        } else {
092            return (Long) response.getFields().get("timestamp").getValue();
093        }
094    }
095
096    /**
097     * Perform bulk indexing
098     *
099     * @param request the index request
100     * @param channel the interactive channel
101     * @throws IOException
102     */
103    public void indexBulk(IndexRequest request, InteractiveChannel channel) throws IOException {
104        bulkHandler.add(request, channel);
105    }
106
107    /**
108     * Perform bulk delete
109     *
110     * @param request the delete request
111     * @param channel the interactive channel
112     * @throws IOException
113     */
114    public void deleteBulk(DeleteRequest request, InteractiveChannel channel) throws IOException {
115        bulkHandler.add(request, channel);
116    }
117
118    /**
119     * Flush bulk
120     *
121     * @param channel the interactive channel
122     * @throws IOException
123     */
124    public void flushBulk(InteractiveChannel channel) throws IOException {
125        bulkHandler.flush();
126    }
127}
