020package org.elasticsearch.websocket.transport.netty;
021
022import java.io.IOException;
023import java.io.NotSerializableException;
024import org.elasticsearch.common.io.ThrowableObjectOutputStream;
025import org.elasticsearch.common.io.stream.Streamable;
026import org.elasticsearch.websocket.common.io.stream.BytesStreamOutput;
027import org.elasticsearch.websocket.common.io.stream.CachedStreamOutput;
028import org.elasticsearch.websocket.transport.support.TransportStreams;
029import org.elasticsearch.transport.NotSerializableTransportException;
030import org.elasticsearch.transport.RemoteTransportException;
031import org.elasticsearch.transport.TransportChannel;
032import org.elasticsearch.transport.TransportResponseOptions;
033import org.jboss.netty.buffer.ChannelBuffer;
034import org.jboss.netty.channel.Channel;
035import org.jboss.netty.channel.ChannelFuture;
036
037/**
038 *
039 */
040public class NettyTransportChannel implements TransportChannel {
041
042    private static final byte[] LENGTH_PLACEHOLDER = new byte[4];
043
044    private final NettyTransport transport;
045
046    private final String action;
047
048    private final Channel channel;
049
050    private final long requestId;
051
052    public NettyTransportChannel(NettyTransport transport, String action, Channel channel, long requestId) {
053        this.transport = transport;
054        this.action = action;
055        this.channel = channel;
056        this.requestId = requestId;
057    }
058
059    @Override
060    public String action() {
061        return this.action;
062    }
063
064    @Override
065    public void sendResponse(Streamable message) throws IOException {
066        sendResponse(message, TransportResponseOptions.EMPTY);
067    }
068
069    @Override
070    public void sendResponse(Streamable message, TransportResponseOptions options) throws IOException {
071        if (transport.compress) {
072            options.withCompress(true);
073        }
074        CachedStreamOutput.Entry cachedEntry = CachedStreamOutput.popEntry();
075        TransportStreams.buildResponse(cachedEntry, requestId, message, options);
076        ChannelBuffer buffer = cachedEntry.bytes().ourBytes().toChannelBuffer();
077        ChannelFuture future = channel.write(buffer);
078        future.addListener(new NettyTransport.CacheFutureListener(cachedEntry));
079    }
080
081    @Override
082    public void sendResponse(Throwable error) throws IOException {
083        CachedStreamOutput.Entry cachedEntry = CachedStreamOutput.popEntry();
084        BytesStreamOutput stream;
085        try {
086            stream = cachedEntry.bytes();
087            writeResponseExceptionHeader(stream);
088            RemoteTransportException tx = new RemoteTransportException(transport.nodeName(), transport.wrapAddress(channel.getLocalAddress()), action, error);
089            ThrowableObjectOutputStream too = new ThrowableObjectOutputStream(stream);
090            too.writeObject(tx);
091            too.close();
092        } catch (NotSerializableException e) {
093            cachedEntry.reset();
094            stream = cachedEntry.bytes();
095            writeResponseExceptionHeader(stream);
096            RemoteTransportException tx = new RemoteTransportException(transport.nodeName(), transport.wrapAddress(channel.getLocalAddress()), action, new NotSerializableTransportException(error));
097            ThrowableObjectOutputStream too = new ThrowableObjectOutputStream(stream);
098            too.writeObject(tx);
099            too.close();
100        }
101        ChannelBuffer buffer = stream.ourBytes().toChannelBuffer();
102        buffer.setInt(0, buffer.writerIndex() - 4); // update real size.
103        ChannelFuture future = channel.write(buffer);
104        future.addListener(new NettyTransport.CacheFutureListener(cachedEntry));
105    }
106
107    private void writeResponseExceptionHeader(BytesStreamOutput stream) throws IOException {
108        stream.writeBytes(LENGTH_PLACEHOLDER);
109        stream.writeLong(requestId);
110        byte status = 0;
111        status = TransportStreams.statusSetResponse(status);
112        status = TransportStreams.statusSetError(status);
113        stream.writeByte(status);
114    }
115}
