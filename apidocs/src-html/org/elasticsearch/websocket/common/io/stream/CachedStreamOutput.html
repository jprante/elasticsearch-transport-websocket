020package org.elasticsearch.websocket.common.io.stream;
021
022import java.io.IOException;
023import java.lang.ref.SoftReference;
024import java.util.Queue;
025import java.util.concurrent.atomic.AtomicInteger;
026import org.elasticsearch.common.compress.Compressor;
027import org.elasticsearch.common.io.UTF8StreamWriter;
028import org.elasticsearch.common.io.stream.HandlesStreamOutput;
029import org.elasticsearch.common.io.stream.StreamOutput;
030import org.elasticsearch.common.unit.ByteSizeValue;
031import org.elasticsearch.common.util.concurrent.ConcurrentCollections;
032import org.elasticsearch.monitor.jvm.JvmInfo;
033
034/**
035 *
036 */
037public class CachedStreamOutput {
038
039    private static Entry newEntry() {
040        BytesStreamOutput bytes = new BytesStreamOutput();
041        HandlesStreamOutput handles = new HandlesStreamOutput(bytes);
042        return new Entry(bytes, handles);
043    }
044
045    public static class Entry {
046        private final BytesStreamOutput bytes;
047        private final HandlesStreamOutput handles;
048
049        Entry(BytesStreamOutput bytes, HandlesStreamOutput handles) {
050            this.bytes = bytes;
051            this.handles = handles;
052        }
053
054        public void reset() {
055            bytes.reset();
056            handles.setOut(bytes);
057            handles.clear();
058        }
059
060        public BytesStreamOutput bytes() {
061            return bytes;
062        }
063
064        public StreamOutput handles() throws IOException {
065            return handles;
066        }
067
068        public StreamOutput bytes(Compressor compressor) throws IOException {
069            return compressor.streamOutput(bytes);
070        }
071
072        public StreamOutput handles(Compressor compressor) throws IOException {
073            StreamOutput compressed = compressor.streamOutput(bytes);
074            handles.clear();
075            handles.setOut(compressed);
076            return handles;
077        }
078    }
079
080    static class SoftWrapper<T> {
081        private SoftReference<T> ref;
082
083        public SoftWrapper() {
084        }
085
086        public void set(T ref) {
087            this.ref = new SoftReference<T>(ref);
088        }
089
090        public T get() {
091            return ref == null ? null : ref.get();
092        }
093
094        public void clear() {
095            ref = null;
096        }
097    }
098
099    private static final SoftWrapper<Queue<Entry>> cache = new SoftWrapper<Queue<Entry>>();
100    private static final AtomicInteger counter = new AtomicInteger();
101    public static int BYTES_LIMIT = 1 * 1024 * 1024; // don't cache entries that are bigger than that...
102    public static int COUNT_LIMIT = 100; // number of concurrent entries cached
103
104    static {
105        // guess the maximum size per entry and the maximum number of entries based on the heap size
106        long maxHeap = JvmInfo.jvmInfo().mem().heapMax().bytes();
107        if (maxHeap < ByteSizeValue.parseBytesSizeValue("500mb").bytes()) {
108            BYTES_LIMIT = (int) ByteSizeValue.parseBytesSizeValue("500kb").bytes();
109            COUNT_LIMIT = 10;
110        } else if (maxHeap < ByteSizeValue.parseBytesSizeValue("1gb").bytes()) {
111            BYTES_LIMIT = (int) ByteSizeValue.parseBytesSizeValue("1mb").bytes();
112            COUNT_LIMIT = 20;
113        } else if (maxHeap < ByteSizeValue.parseBytesSizeValue("4gb").bytes()) {
114            BYTES_LIMIT = (int) ByteSizeValue.parseBytesSizeValue("2mb").bytes();
115            COUNT_LIMIT = 50;
116        } else if (maxHeap < ByteSizeValue.parseBytesSizeValue("10gb").bytes()) {
117            BYTES_LIMIT = (int) ByteSizeValue.parseBytesSizeValue("5mb").bytes();
118            COUNT_LIMIT = 50;
119        } else {
120            BYTES_LIMIT = (int) ByteSizeValue.parseBytesSizeValue("10mb").bytes();
121            COUNT_LIMIT = 100;
122        }
123    }
124
125    public static void clear() {
126        cache.clear();
127    }
128
129    public static Entry popEntry() {
130        Queue<Entry> ref = cache.get();
131        if (ref == null) {
132            return newEntry();
133        }
134        Entry entry = ref.poll();
135        if (entry == null) {
136            return newEntry();
137        }
138        counter.decrementAndGet();
139        entry.reset();
140        return entry;
141    }
142
143    public static void pushEntry(Entry entry) {
144        entry.reset();
145        if (entry.bytes().bytes().length() > BYTES_LIMIT) {
146            return;
147        }
148        Queue<Entry> ref = cache.get();
149        if (ref == null) {
150            ref = ConcurrentCollections.newQueue();
151            counter.set(0);
152            cache.set(ref);
153        }
154        if (counter.incrementAndGet() > COUNT_LIMIT) {
155            counter.decrementAndGet();
156        } else {
157            ref.add(entry);
158        }
159    }
160
161    private static ThreadLocal<SoftReference<UTF8StreamWriter>> utf8StreamWriter = new ThreadLocal<SoftReference<UTF8StreamWriter>>();
162
163    public static UTF8StreamWriter utf8StreamWriter() {
164        SoftReference<UTF8StreamWriter> ref = utf8StreamWriter.get();
165        UTF8StreamWriter writer = (ref == null) ? null : ref.get();
166        if (writer == null) {
167            writer = new UTF8StreamWriter(1024 * 4);
168            utf8StreamWriter.set(new SoftReference<UTF8StreamWriter>(writer));
169        }
170        writer.reset();
171        return writer;
172    }
173}
