019package org.elasticsearch.websocket.http.netty;
020
021import org.elasticsearch.websocket.Presence;
022import org.jboss.netty.channel.ChannelHandler;
023import org.jboss.netty.channel.ChannelHandlerContext;
024import org.jboss.netty.channel.ExceptionEvent;
025import org.jboss.netty.channel.MessageEvent;
026import org.jboss.netty.channel.SimpleChannelUpstreamHandler;
027import org.jboss.netty.handler.codec.http.HttpHeaders;
028import org.jboss.netty.handler.codec.http.HttpRequest;
029import org.jboss.netty.handler.codec.http.websocketx.WebSocketServerHandshaker;
030import org.jboss.netty.handler.codec.http.websocketx.WebSocketServerHandshakerFactory;
031
032/**
033 *  Handles HTTP request and upgrades HTTP to WebSocket if appropriate.
034 * 
035 * 
036 */
037@ChannelHandler.Sharable
038public class NettyHttpRequestHandler extends SimpleChannelUpstreamHandler {
039
040    protected final NettyWebSocketServerTransport serverTransport;
041    protected WebSocketServerHandshaker handshaker;
042
043    private static String getWebSocketLocation(HttpRequest req) {
044        return "ws://" + req.getHeader(HttpHeaders.Names.HOST) + WEBSOCKET_PATH;
045    }
046    private static final String WEBSOCKET_PATH = "/websocket";
047    
048    public NettyHttpRequestHandler(NettyWebSocketServerTransport serverTransport) {
049        this.serverTransport = serverTransport;
050    }
051
052    @Override
053    public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
054        HttpRequest request = (HttpRequest) e.getMessage();
055        if (request.getUri().startsWith(WEBSOCKET_PATH)) {
056            // Websocket handshake
057            WebSocketServerHandshakerFactory wsFactory = new WebSocketServerHandshakerFactory(getWebSocketLocation(request), null, false);
058            handshaker = wsFactory.newHandshaker(request);
059            if (handshaker == null) {
060                wsFactory.sendUnsupportedWebSocketVersionResponse(ctx.getChannel());
061            } else {
062                handshaker.handshake(ctx.getChannel(), request).addListener(WebSocketServerHandshaker.HANDSHAKE_LISTENER);                
063            }
064            // extract topic from request URI
065            String topic = request.getUri();
066            if (topic.length() > WEBSOCKET_PATH.length() + 1) {
067                topic = topic.substring(WEBSOCKET_PATH.length() + 1);
068                serverTransport.presence(Presence.CONNECTED, topic, e.getChannel());
069            }
070        } else {
071            serverTransport.dispatchRequest(new NettyHttpRequest(request), new NettyHttpChannel(serverTransport, e.getChannel(), request));
072            super.messageReceived(ctx, e);
073        }
074    }
075
076    @Override
077    public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
078        serverTransport.exceptionCaught(ctx, e);
079        e.getChannel().close();
080    }
081
082}
