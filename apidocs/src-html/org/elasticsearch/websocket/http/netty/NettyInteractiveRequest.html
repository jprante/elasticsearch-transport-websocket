019package org.elasticsearch.websocket.http.netty;
020
021import java.io.IOException;
022import java.util.Map;
023import org.elasticsearch.common.unit.TimeValue;
024import org.elasticsearch.common.xcontent.XContentBuilder;
025import org.elasticsearch.websocket.InteractiveRequest;
026import org.elasticsearch.websocket.client.WebSocketClient;
027import org.elasticsearch.websocket.client.WebSocketClientRequest;
028
029/**
030 * Netty implemenation for an interactive request.
031 * At the same time, this class serves as an implementation
032 * for a WebSocket client request.
033 * 
034 * @author JÃ¶rg Prante <joergprante@gmail.com>
035 */
036public class NettyInteractiveRequest implements InteractiveRequest, WebSocketClientRequest {
037
038    protected String type; 
039    
040    protected Map<String, Object> data;    
041
042    protected XContentBuilder builder;
043    
044    public NettyInteractiveRequest() {
045    }
046    
047    public NettyInteractiveRequest(Map<String, Object> data) {
048        this.data = data;
049    }
050    
051    @Override
052    public NettyInteractiveRequest type(String type) {
053        this.type = type;
054        return this;
055    }
056    
057    @Override
058    public NettyInteractiveRequest data(XContentBuilder builder) {
059        this.builder = builder;
060        return this;
061    }
062    
063    @Override
064    public void send(WebSocketClient client) throws IOException {
065        client.send(new NettyInteractiveResponse(type, builder).response());
066    }
067
068    @Override
069    public Map<String, Object> asMap() {
070        return data;
071    }
072
073    @Override
074    public boolean hasParam(String key) {
075        return data.containsKey(key);
076    }
077
078    @Override
079    public Object param(String key) {
080        return data.get(key);
081    }
082
083    @Override
084    public String paramAsString(String key) {
085        Object o = param(key);
086        return o != null ? o.toString() : null;
087    }
088
089    @Override
090    public String paramAsString(String key, String defaultValue) {
091        Object o = param(key);
092        return o != null ? o.toString() : defaultValue;
093    }
094
095    @Override
096    public long paramAsLong(String key) {
097        Object o = param(key);
098        try {
099            return o != null ? Long.parseLong(o.toString()) : null;
100        } catch (NumberFormatException e) {
101            return 0L;
102        }
103    }
104
105    @Override
106    public long paramAsLong(String key, long defaultValue) {
107        Object o = param(key);
108        try {
109            return o != null ? Long.parseLong(o.toString()) : defaultValue;
110        } catch (NumberFormatException e) {
111            return defaultValue;
112        }
113    }
114
115    @Override
116    public boolean paramAsBoolean(String key) {
117        Object o = param(key);
118        return o != null ? Boolean.getBoolean(o.toString()) : null;
119    }
120
121    @Override
122    public boolean paramAsBoolean(String key, boolean defaultValue) {
123        Object o = param(key);
124        return o != null ? Boolean.getBoolean(o.toString()) : defaultValue;
125    }
126
127    @Override
128    public TimeValue paramAsTime(String key) {
129        Object o = param(key);
130        return o != null ? TimeValue.parseTimeValue(o.toString(), null) : null;
131    }
132
133    @Override
134    public TimeValue paramAsTime(String key, TimeValue defaultValue) {
135        Object o = param(key);
136        return o != null ? TimeValue.parseTimeValue(o.toString(), defaultValue) : defaultValue;
137    }
138}
