020package org.elasticsearch.websocket.client;
021
022import org.jboss.netty.channel.Channel;
023import org.jboss.netty.channel.ChannelFuture;
024import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
025
026/**
027 * A WebSocket client.
028 *
029 * @author <a href="http://www.pedantique.org/">Carl Bystr&ouml;m</a>
030 * @author JÃ¶rg Prante <joergprante@gmail.com>
031 */
032public interface WebSocketClient<F extends ChannelFuture, Frame extends WebSocketFrame> {
033
034    /**
035     * The channel this client has opened.
036     * @return the channel
037     */
038    Channel channel();
039    
040    /**
041     * Connect to host and port.
042     *
043     * @return Connect future. Fires when connected.
044     */
045    F connect();
046
047    /**
048     * Disconnect from the server
049     *
050     * @return Disconnect future. Fires when disconnected.
051     */
052    F disconnect();
053
054    /**
055     * Send data to server
056     *
057     * @param frame Data for sending
058     * @return Write future. Will fire when the data is sent.
059     */
060    F send(Frame frame);
061}
