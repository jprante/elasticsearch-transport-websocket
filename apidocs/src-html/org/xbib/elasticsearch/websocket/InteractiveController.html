002    package org.xbib.elasticsearch.websocket;
003    
004    import java.nio.charset.Charset;
005    import java.util.HashMap;
006    import java.util.Map;
007    
008    import org.elasticsearch.common.logging.ESLogger;
009    import org.elasticsearch.common.logging.ESLoggerFactory;
010    import org.jboss.netty.channel.Channel;
011    import org.jboss.netty.channel.ChannelHandlerContext;
012    import org.jboss.netty.handler.codec. http.websocketx.CloseWebSocketFrame;
013    import org.jboss.netty.handler.codec.http.websocketx.PingWebSocketFrame;
014    import org.jboss.netty.handler.codec.http.websocketx.PongWebSocketFrame;
015    import org.jboss.netty.handler.codec.http.websocketx.TextWebSocketFrame;
016    import org.jboss.netty.handler.codec.http.websocketx.WebSocketFrame;
017    import org.jboss.netty.handler.codec.http.websocketx.WebSocketServerHandshaker;
018    
019    import org.elasticsearch.ElasticsearchException;
020    import org.elasticsearch.common.component.AbstractLifecycleComponent;
021    import org.elasticsearch.common.inject.Inject;
022    import org.elasticsearch.common.settings.Settings;
023    import org.elasticsearch.common.xcontent.XContentFactory;
024    import org.elasticsearch.common.xcontent.XContentParser;
025    import org.xbib.elasticsearch.websocket.http.netty.NettyInteractiveChannel;
026    import org.xbib.elasticsearch.websocket.http.netty.NettyInteractiveRequest;
027    
028    /**
029     * The InteractiveController controls the presence of websocket connections
030     * and the flow of websocket frames for interactive use.
031     */
032    public class InteractiveController extends AbstractLifecycleComponent<InteractiveController> {
033    
034        private final ESLogger logger = ESLoggerFactory.getLogger(InteractiveController.class.getSimpleName());
035    
036        private final HashMap<String, InteractiveHandler> handlers = new HashMap();
037    
038        @Inject
039        public InteractiveController(Settings settings) {
040            super(settings);
041        }
042    
043        @Override
044        protected void doStart() throws ElasticsearchException {
045        }
046    
047        @Override
048        protected void doStop() throws ElasticsearchException {
049        }
050    
051        @Override
052        protected void doClose() throws ElasticsearchException {
053        }
054    
055        public void registerHandler(String type, InteractiveHandler handler) {
056            handlers.put(type, handler);
057        }
058    
059        public void presence(Presence presence, String topic, Channel channel) {
060            if (logger.isDebugEnabled()) {
061                logger.debug("presence: " + presence.name()
062                        + " topic =" + topic
063                        + " channel =" + channel);
064            }
065        }
066    
067        public void frame(WebSocketServerHandshaker handshaker, WebSocketFrame frame, ChannelHandlerContext context) {
068            Channel channel = context.getChannel();
069            if (frame instanceof TextWebSocketFrame) {
070                text((TextWebSocketFrame) frame,channel);
071            } else if (handshaker != null && frame instanceof CloseWebSocketFrame) {
072                handshaker.close(context.getChannel(), (CloseWebSocketFrame) frame);
073                presence(Presence.DISCONNECTED, null, channel);
074            } else if (frame instanceof PingWebSocketFrame) {
075                channel.write(new PongWebSocketFrame(frame.getBinaryData()));
076            }
077        }
078    
079        private void text(TextWebSocketFrame frame, Channel channel) {
080            Map<String, Object> map = parse(frame.getBinaryData().toString(Charset.forName("UTF-8")));
081            if (map == null) {
082                error("invalid request", channel);
083                return;
084            }
085            String type = (String) map.get("type");
086            if (type == null) {
087                error("no type found", channel);
088                return;
089            }
090            if (!handlers.containsKey(type)) {
091                error("missing handler for type: " + type, channel);
092                return;
093            }
094            Map<String, Object> data = (Map<String, Object>) map.get("data");
095            handlers.get(type).handleRequest(new NettyInteractiveRequest(data),
096                    new NettyInteractiveChannel(channel));
097        }
098    
099        private Map<String, Object> parse(String source) {
100            Map<String, Object> map = null;
101            XContentParser parser = null;
102            try {
103                parser = XContentFactory.xContent(source).createParser(source);
104                map = parser.map();
105            } catch (Exception e) {
106                logger.error("unable to parse: {}", source);
107            } finally {
108                if (parser != null) {
109                    parser.close();
110                }
111            }
112            return map;
113        }
114    
115        private void error(String message, Channel channel) {
116            String text = "{\"ok\":false,\"error\":\"" + message + "\"}";
117            TextWebSocketFrame frame = new TextWebSocketFrame(text);
118            channel.write(frame);
119        }
120    }
