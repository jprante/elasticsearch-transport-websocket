123        @Inject
124        public NettyWebSocketServerTransport(Settings settings, NetworkService networkService, Checkpointer pubsubService) {
125            super(settings);
126            this.networkService = networkService;
127            ByteSizeValue maxContentLength = componentSettings.getAsBytesSize("max_content_length", settings.getAsBytesSize("websocket.max_content_length", new ByteSizeValue(100, ByteSizeUnit.MB)));
128            this.maxChunkSize = componentSettings.getAsBytesSize("max_chunk_size", settings.getAsBytesSize("websocket.max_chunk_size", new ByteSizeValue(8, ByteSizeUnit.KB)));
129            this.maxHeaderSize = componentSettings.getAsBytesSize("max_header_size", settings.getAsBytesSize("websocket.max_header_size", new ByteSizeValue(8, ByteSizeUnit.KB)));
130            this.maxInitialLineLength = componentSettings.getAsBytesSize("max_initial_line_length", settings.getAsBytesSize("websocket.max_initial_line_length", new ByteSizeValue(4, ByteSizeUnit.KB)));
131            // don't reset cookies by default, since I don't think we really need to
132            // note, parsing cookies was fixed in netty 3.5.1 regarding stack allocation, but still, currently, we don't need cookies
133            this.resetCookies = componentSettings.getAsBoolean("reset_cookies", settings.getAsBoolean("websocket.reset_cookies", false));
134            this.maxCumulationBufferCapacity = componentSettings.getAsBytesSize("max_cumulation_buffer_capacity", null);
135            this.maxCompositeBufferComponents = componentSettings.getAsInt("max_composite_buffer_components", -1);
136            this.workerCount = componentSettings.getAsInt("worker_count", Runtime.getRuntime().availableProcessors() * 2);
137            this.port = componentSettings.get("port", settings.get("websocket.port", "9400-9500"));
138            this.bindHost = componentSettings.get("bind_host", settings.get("websocket.bind_host", settings.get("websocket.host")));
139            this.publishHost = componentSettings.get("publish_host", settings.get("websocket.publish_host", settings.get("websocket.host")));
140            this.tcpNoDelay = componentSettings.getAsBoolean("tcp_no_delay", settings.getAsBoolean(TCP_NO_DELAY, true));
141            this.tcpKeepAlive = componentSettings.getAsBoolean("tcp_keep_alive", settings.getAsBoolean(TCP_KEEP_ALIVE, true));
142            this.reuseAddress = componentSettings.getAsBoolean("reuse_address", settings.getAsBoolean(TCP_REUSE_ADDRESS, NetworkUtils.defaultReuseAddress()));
143            this.tcpSendBufferSize = componentSettings.getAsBytesSize("tcp_send_buffer_size", settings.getAsBytesSize(TCP_SEND_BUFFER_SIZE, TCP_DEFAULT_SEND_BUFFER_SIZE));
144            this.tcpReceiveBufferSize = componentSettings.getAsBytesSize("tcp_receive_buffer_size", settings.getAsBytesSize(TCP_RECEIVE_BUFFER_SIZE, TCP_DEFAULT_RECEIVE_BUFFER_SIZE));
145    
146            long defaultReceiverPredictor = 512 * 1024;
147            // skip JVM info
148            ByteSizeValue receivePredictorMin = componentSettings.getAsBytesSize("receive_predictor_min", componentSettings.getAsBytesSize("receive_predictor_size", new ByteSizeValue(defaultReceiverPredictor)));
149            ByteSizeValue receivePredictorMax = componentSettings.getAsBytesSize("receive_predictor_max", componentSettings.getAsBytesSize("receive_predictor_size", new ByteSizeValue(defaultReceiverPredictor)));
150            if (receivePredictorMax.bytes() == receivePredictorMin.bytes()) {
151                receiveBufferSizePredictorFactory = new FixedReceiveBufferSizePredictorFactory((int) receivePredictorMax.bytes());
152            } else {
153                receiveBufferSizePredictorFactory = new AdaptiveReceiveBufferSizePredictorFactory((int) receivePredictorMin.bytes(), (int) receivePredictorMin.bytes(), (int) receivePredictorMax.bytes());
154            }
155    
156            this.compression = settings.getAsBoolean("websocket.compression", false);
157            this.compressionLevel = settings.getAsInt("websocket.compression_level", 6);
158    
159            // validate max content length
160            if (maxContentLength.bytes() > Integer.MAX_VALUE) {
161                logger.warn("maxContentLength[" + maxContentLength + "] set to high value, resetting it to [100mb]");
162                maxContentLength = new ByteSizeValue(100, ByteSizeUnit.MB);
163            }
164            this.maxContentLength = maxContentLength;
165    
166            logger.debug("using max_chunk_size[{}], max_header_size[{}], max_initial_line_length[{}], max_content_length[{}]",
167                    maxChunkSize, maxHeaderSize, maxInitialLineLength, this.maxContentLength);
168        }

181        @Override
182        protected void doStart() throws ElasticsearchException {
183            this.serverOpenChannels = new OpenChannelsHandler(logger);
184    
185            /* we do not support oio for websocket - it wouldn't work either */
186            serverBootstrap = new ServerBootstrap(new NioServerSocketChannelFactory(
187                    Executors.newCachedThreadPool(daemonThreadFactory(settings, "websocket_server_boss")),
188                    Executors.newCachedThreadPool(daemonThreadFactory(settings, "websocket_server_worker")),
189                    workerCount));
190    
191            serverBootstrap.setPipelineFactory(new NettyWebSocketServerPipelineFactory(this));
192    
193            if (tcpNoDelay != null) {
194                serverBootstrap.setOption("child.tcpNoDelay", tcpNoDelay);
195            }
196            if (tcpKeepAlive != null) {
197                serverBootstrap.setOption("child.keepAlive", tcpKeepAlive);
198            }
199            if (tcpSendBufferSize != null && tcpSendBufferSize.bytes() > 0) {
200                serverBootstrap.setOption("child.sendBufferSize", tcpSendBufferSize.bytes());
201            }
202            if (tcpReceiveBufferSize != null && tcpReceiveBufferSize.bytes() > 0) {
203                serverBootstrap.setOption("child.receiveBufferSize", tcpReceiveBufferSize.bytes());
204            }
205            serverBootstrap.setOption("receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
206            serverBootstrap.setOption("child.receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
207            if (reuseAddress != null) {
208                serverBootstrap.setOption("reuseAddress", reuseAddress);
209                serverBootstrap.setOption("child.reuseAddress", reuseAddress);
210            }
211    
212            // Bind and start to accept incoming connections.
213            InetAddress hostAddressX;
214            try {
215                hostAddressX = networkService.resolveBindHostAddress(bindHost);
216            } catch (IOException e) {
217                throw new BindHttpException("Failed to resolve host [" + bindHost + "]", e);
218            }
219            final InetAddress hostAddress = hostAddressX;
220    
221            PortsRange portsRange = new PortsRange(port);
222            final AtomicReference<Exception> lastException = new AtomicReference();
223            boolean success = portsRange.iterate(new PortsRange.PortCallback() {
224                @Override
225                public boolean onPortNumber(int portNumber) {
226                    try {
227                        serverChannel = serverBootstrap.bind(new InetSocketAddress(hostAddress, portNumber));
228                    } catch (Exception e) {
229                        lastException.set(e);
230                        return false;
231                    }
232                    return true;
233                }
234            });
235            if (!success) {
236                throw new BindHttpException("Failed to bind to [" + port + "]", lastException.get());
237            }
238            InetSocketAddress boundAddress = (InetSocketAddress) serverChannel.getLocalAddress();
239            InetSocketAddress publishAddress;
240            try {
241                publishAddress = new InetSocketAddress(networkService.resolvePublishHostAddress(publishHost), boundAddress.getPort());
242            } catch (Exception e) {
243                throw new BindTransportException("Failed to resolve publish address", e);
244            }
245            this.boundAddress = new BoundTransportAddress(new InetSocketTransportAddress(boundAddress), new InetSocketTransportAddress(publishAddress));
246        }
247        @Override
248        protected void doStop() throws ElasticsearchException {
249            if (serverChannel != null) {
250                serverChannel.close().awaitUninterruptibly();
251                serverChannel = null;
252            }
253            if (serverOpenChannels != null) {
254                serverOpenChannels.close();
255                serverOpenChannels = null;
256            }
257        }

317        /**
318         * Forward a message to another node with websockets for delivery.
319         *
320         * @param websocketNodeAddress the websocket address of the other node, e.g.
321         * "/10.0.0.1:9400"
322         * @param channelId the channel ID on the other node for delivering the
323         * message
324         * @param builder the builder for the message
325         */
326        @Override
327        public void forward(final String websocketNodeAddress, final Integer channelId, final XContentBuilder builder) {
328            try {
329                // build "forward" text frame
330                XContentBuilder forwardBuilder = jsonBuilder();
331                forwardBuilder.startObject()
332                        .field("channel", channelId)
333                        .rawField("message", builder.bytes())
334                        .endObject();
335                final TextWebSocketFrame frame = new NettyInteractiveResponse("forward", forwardBuilder).response();
336                // use a websocket client pool
337                WebSocketClient client = nodeChannels.get(websocketNodeAddress);
338                if (client == null) {
339                    final URI uri = new URI("ws:/" + websocketNodeAddress + "/websocket");
340                    client = clientFactory.newClient(uri, new WebSocketActionListener() {
341                        @Override
342                        public void onConnect(WebSocketClient client) {
343                            nodeChannels.put(websocketNodeAddress, client);
344                            client.send(frame);
345                        }
346    
347                        @Override
348                        public void onDisconnect(WebSocketClient client) {
349                            logger.warn("node disconnected: {}", uri);
350                            nodeChannels.remove(websocketNodeAddress);
351                        }
352    
353                        @Override
354                        public void onMessage(WebSocketClient client, WebSocketFrame frame) {
355                            logger.info("unexpected response {}", frame);
356                        }
357    
358                        @Override
359                        public void onError(Throwable t) {
360                            logger.error(t.getMessage(), t);
361                            nodeChannels.remove(websocketNodeAddress);
362                        }
363                    });
364                    client.connect();
365                } else {
366                    client.send(frame);
367                }
368            } catch (Exception e) {
369                logger.error(e.getMessage(), e);
370            }
371        }
372    
373        void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
374            if (e.getCause() instanceof ReadTimeoutException) {
375                if (logger.isTraceEnabled()) {
376                    logger.trace("Connection timeout [{}]", ctx.getChannel().getRemoteAddress());
377                }
378                ctx.getChannel().close();
379            } else {
380                if (!lifecycle.started()) {
381                    // ignore
382                    return;
383                }
384                if (!NetworkExceptionHelper.isCloseConnectionException(e.getCause())) {
385                    logger.warn("Caught exception while handling client http traffic, closing connection {}", e.getCause(), ctx.getChannel());
386                    ctx.getChannel().close();
387                } else {
388                    logger.debug("Caught exception while handling client http traffic, closing connection {}", e.getCause(), ctx.getChannel());
389                    ctx.getChannel().close();
390                }
391            }
392        }
