001    
002    package org.xbib.elasticsearch.websocket.http;
003    
004    import org.elasticsearch.common.collect.ImmutableList;
005    import org.elasticsearch.common.inject.AbstractModule;
006    import org.elasticsearch.common.inject.Module;
007    import org.elasticsearch.common.inject.Modules;
008    import org.elasticsearch.common.inject.SpawnModules;
009    import org.elasticsearch.common.settings.Settings;
010    
011    import org.xbib.elasticsearch.websocket.http.netty.NettyWebSocketServerTransportModule;
012    
013    public class HttpServerModule extends AbstractModule implements SpawnModules {
014    
015        private final Settings settings;
016    
017        public HttpServerModule(Settings settings) {
018            this.settings = settings;
019        }
020    
021        @Override
022        public Iterable<? extends Module> spawnModules() {
023            return ImmutableList.of(Modules.createModule(settings.getAsClass("websocket.type", NettyWebSocketServerTransportModule.class, "org.elasticsearch.websocket.http.", "HttpServerTransportModule"), settings));
024        }
025    
026        @SuppressWarnings({"unchecked"})
027        @Override
028        protected void configure() {
029            bind(HttpServer.class).asEagerSingleton();
030        }
031    }
