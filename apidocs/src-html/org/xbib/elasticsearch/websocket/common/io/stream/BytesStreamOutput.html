002    package org.xbib.elasticsearch.websocket.common.io.stream;
003    
004    import java.io.IOException;
005    import java.util.Arrays;
006    
007    import org.apache.lucene.util.ArrayUtil;
008    
009    import org.elasticsearch.common.io.BytesStream;
010    import org.elasticsearch.common.io.stream.StreamOutput;
011    import org.xbib.elasticsearch.websocket.common.bytes.BytesArray;
012    import org.xbib.elasticsearch.websocket.common.bytes.BytesReference;
013    
014    public class BytesStreamOutput extends StreamOutput implements BytesStream {
015    
016        /**
017         * The buffer where data is stored.
018         */
019        protected byte buf[];
020    
021        /**
022         * The number of valid bytes in the buffer.
023         */
024        protected int count;
025    
026        public BytesStreamOutput() {
027            this(1024);
028        }
029    
030        public BytesStreamOutput(int size) {
031            this.buf = new byte[size];
032        }
033    
034        @Override
035        public boolean seekPositionSupported() {
036            return true;
037        }
038    
039        @Override
040        public long position() throws IOException {
041            return count;
042        }
043    
044        @Override
045        public void seek(long position) throws IOException {
046            if (position > Integer.MAX_VALUE) {
047                throw new UnsupportedOperationException();
048            }
049            count = (int) position;
050        }
051    
052        @Override
053        public void writeByte(byte b) throws IOException {
054            int newcount = count + 1;
055            if (newcount > buf.length) {
056                buf = Arrays.copyOf(buf, ArrayUtil.oversize(newcount, 1));
057            }
058            buf[count] = b;
059            count = newcount;
060        }
061    
062        public void skip(int length) {
063            int newcount = count + length;
064            if (newcount > buf.length) {
065                buf = grow(newcount);
066            }
067            count = newcount;
068        }
069    
070        private byte[] grow(int newCount) {
071            // try and grow faster while we are small...
072            if (newCount < 256 * 1024) {
073                newCount = Math.max(buf.length << 1, newCount);
074            }
075            return ArrayUtil.grow(buf, newCount);
076        }
077    
078        @Override
079        public void writeBytes(byte[] b, int offset, int length) throws IOException {
080            if (length == 0) {
081                return;
082            }
083            int newcount = count + length;
084            if (newcount > buf.length) {
085                buf = Arrays.copyOf(buf, ArrayUtil.oversize(newcount, 1));
086            }
087            System.arraycopy(b, offset, buf, count, length);
088            count = newcount;
089        }
090    
091        public void seek(int seekTo) {
092            count = seekTo;
093        }
094    
095        @Override
096        public void reset() {
097            count = 0;
098        }
099    
100        @Override
101        public void flush() throws IOException {
102            // nothing to do there
103        }
104    
105        @Override
106        public void close() throws IOException {
107            // nothing to do here
108        }
109    
110        /**
111         * This method needs to be substituted to return our un-shaded Netty BytesReference
112         * @return throws exception
113         */
114        @Override
115        public org.elasticsearch.common.bytes.BytesReference bytes() {
116            throw new UnsupportedOperationException("Not supported yet.");
117        }
118    
119        /**
120         * Only this bytes() substitution method must be used.
121         * @return bytes array
122         */
123        public BytesReference ourBytes() {
124            return new BytesArray(buf, 0, count);
125        }
126    
127        /**
128         * Returns the current size of the buffer.
129         *
130         * @return the value of the <code>count</code> field, which is the number
131         *         of valid bytes in this output stream.
132         * @see java.io.ByteArrayOutputStream#count
133         */
134        public int size() {
135            return count;
136        }
137    
138    }
