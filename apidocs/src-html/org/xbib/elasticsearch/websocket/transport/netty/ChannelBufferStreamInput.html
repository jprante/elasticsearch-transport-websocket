002    package org.xbib.elasticsearch.websocket.transport.netty;
003    
004    import java.io.EOFException;
005    import java.io.IOException;
006    
007    import org.elasticsearch.common.bytes.BytesReference;
008    import org.elasticsearch.common.io.stream.StreamInput;
009    
010    import org.jboss.netty.buffer.ChannelBuffer;
011    import org.xbib.elasticsearch.websocket.common.bytes.ChannelBufferBytesReference;
012    
013    /**
014     * A Netty {@link org.jboss.netty.buffer.ChannelBuffer} based {@link org.elasticsearch.common.io.stream.StreamInput}.
015     */
016    public class ChannelBufferStreamInput extends StreamInput {
017    
018        private final ChannelBuffer buffer;
019    
020        private final int startIndex;
021    
022        private final int endIndex;
023    
024        public ChannelBufferStreamInput(ChannelBuffer buffer) {
025            this(buffer, buffer.readableBytes());
026        }
027    
028        public ChannelBufferStreamInput(ChannelBuffer buffer, int length) {
029            if (length > buffer.readableBytes()) {
030                throw new IndexOutOfBoundsException();
031            }
032            this.buffer = buffer;
033            startIndex = buffer.readerIndex();
034            endIndex = startIndex + length;
035            buffer.markReaderIndex();
036        }
037    
038        @Override
039        public BytesReference readBytesReference(int length) throws IOException {
040            ChannelBufferBytesReference ref = new ChannelBufferBytesReference(buffer.slice(buffer.readerIndex(), length));
041            buffer.skipBytes(length);
042            return ref;
043        }
044    
045        @Override
046        public int available() throws IOException {
047            return endIndex - buffer.readerIndex();
048        }
049    
050        @Override
051        public void mark(int readlimit) {
052            buffer.markReaderIndex();
053        }
054    
055        @Override
056        public boolean markSupported() {
057            return true;
058        }
059    
060        @Override
061        public int read() throws IOException {
062            if (available() == 0) {
063                return -1;
064            }
065            return buffer.readByte() & 0xff;
066        }
067    
068        @Override
069        public int read(byte[] b, int off, int len) throws IOException {
070            if (len == 0) {
071                return 0;
072            }
073            int available = available();
074            if (available == 0) {
075                return -1;
076            }
077    
078            len = Math.min(available, len);
079            buffer.readBytes(b, off, len);
080            return len;
081        }
082    
083        @Override
084        public void reset() throws IOException {
085            buffer.resetReaderIndex();
086        }
087    
088        @Override
089        public long skip(long n) throws IOException {
090            if (n > Integer.MAX_VALUE) {
091                return skipBytes(Integer.MAX_VALUE);
092            } else {
093                return skipBytes((int) n);
094            }
095        }
096    
097        public int skipBytes(int n) throws IOException {
098            int nBytes = Math.min(available(), n);
099            buffer.skipBytes(nBytes);
100            return nBytes;
101        }
102    
103    
104        @Override
105        public byte readByte() throws IOException {
106            return buffer.readByte();
107        }
108    
109        @Override
110        public void readBytes(byte[] b, int offset, int len) throws IOException {
111            int read = read(b, offset, len);
112            if (read < len) {
113                throw new EOFException();
114            }
115        }
116    
117        @Override
118        public void close() throws IOException {
119            // nothing to do here
120        }
121    }
