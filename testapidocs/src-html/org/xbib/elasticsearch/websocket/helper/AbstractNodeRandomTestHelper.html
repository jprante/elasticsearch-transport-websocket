001    package org.xbib.elasticsearch.websocket.helper;
002    
003    import java.util.Random;
004    
005    public class AbstractNodeRandomTestHelper extends AbstractNodeTestHelper {
006    
007        private static Random random = new Random();
008    
009        private static char[] numbersAndLetters = ("0123456789abcdefghijklmnopqrstuvwxyz").toCharArray();
010    
011        protected String randomString(int len) {
012            final char[] buf = new char[len];
013            final int n = numbersAndLetters.length - 1;
014            for (int i = 0; i < buf.length; i++) {
015                buf[i] = numbersAndLetters[random.nextInt(n)];
016            }
017            return new String(buf);
018        }
019    
020    }
